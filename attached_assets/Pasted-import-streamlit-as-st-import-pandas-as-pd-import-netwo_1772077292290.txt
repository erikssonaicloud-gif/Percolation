# ====================== PERCOLATION FUNCTIONS (paper-compliant) ======================
def build_percolation_network(events_df, time_window_minutes=30):
    G = nx.Graph()
    df = events_df.copy()
    df['timestamp'] = pd.to_datetime(df['timestamp'])
    df = df.sort_values('timestamp')
    
    for _, row in df.iterrows():
        node_id = row['file_id']
        if row['action'] in ['view', 'open']:
            ntype = 'O'
        elif row['action'] in ['edit', 'save', 'export']:
            ntype = 'A'
        else:
            ntype = 'I'
        G.add_node(node_id, type=ntype, name=row['file_name'], ts=row['timestamp'])
    
    events = df.to_dict('records')
    for i in range(len(events)):
        for j in range(i+1, len(events)):
            diff = events[j]['timestamp'] - events[i]['timestamp']
            if diff > timedelta(minutes=time_window_minutes): break
            weight = 1 / max(diff.total_seconds(), 1)
            G.add_edge(events[i]['file_id'], events[j]['file_id'], weight=weight)
    return G

def calculate_percolation_metrics(G):
    if len(G.nodes()) == 0: return 1.0, 0.0, 0.0
    O_nodes = [n for n,d in G.nodes(data=True) if d['type']=='O']
    A_nodes = [n for n,d in G.nodes(data=True) if d['type']=='A']
    if len(O_nodes)==0: return 0.0, 1.0, 1.0
    if len(A_nodes)==0: return 1.0, 0.0, 0.0
    
    giant = G.subgraph(max(nx.connected_components(G), key=len))
    connected_O = sum(1 for o in O_nodes if any(nx.has_path(giant, o, a) for a in A_nodes))
    kappa = connected_O / len(O_nodes)
    poo = len(giant) / len(G)
    frag = 1.0 - (kappa * poo)
    return frag, kappa, poo
